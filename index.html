def splosni_fibonacci(n, a=0, b=1):

"""Vrne n-ti člen zaporedja a, b, a + b, a + 2 b, …"""

if n == 0:

return a

elif n == 1:

return b

else:

return splosni_fibonacci(n - 1, b, a + b)

def gcd(m, n):

if n == 0:

return m

else:

return gcd(n, m % n)
